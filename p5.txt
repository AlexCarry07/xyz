print("===== Welcome to Admission Chatbot =====")

while True:

    q = input("\nAsk Question: ")
    q = q.lower()

    if "course" in q:
        print("Available courses are CSE, IT, AIDS and ECE")

    elif "fees" in q:
        print("Fees is 1,00,000 per year")

    elif "admission" in q:
        print("Admission is based on entrance exam score")

    elif "hostel" in q:
        print("Hostel facility is available for boys and girls")

    elif "contact" in q:
        print("Contact: 9876543210")

    elif "library" in q:
        print("Library facility is available from 8 AM to 8 PM")

    elif "placement" in q:
        print("Top companies visit for placements every year")

    elif "canteen" in q:
        print("Canteen facility is available inside campus")

    elif "scholarship" in q:
        print("Scholarships are available for eligible students")

    elif "faculty" in q:
        print("Experienced and qualified faculty members are available")

    elif "exam" in q:
        print("Exams are conducted semester wise")

    elif "timing" in q:
        print("College timing is 9 AM to 5 PM")

    elif "location" in q:
        print("College is located in Pune")

    elif "transport" in q:
        print("Bus transport facility is available")

    elif "wifi" in q:
        print("Free WiFi facility is available on campus")

    elif "sports" in q:
        print("Sports and gym facilities are available")

    elif "exit" in q:
        print("Thank You for using Admission Chatbot")
        break

    else:
        print("Sorry, I don't understand")
















































































































































































































































































































































































































































































































































































































































































































































































































This program works like a simple college inquiry chatbot.
It helps users get information about admission and college facilities.
First, a welcome message is displayed.
The chatbot continuously asks the user to enter a question.
User input is converted into lowercase so the chatbot can understand words easily.
The chatbot checks the question using different conditions.
If the question contains words like course, fees, hostel, placement, etc., it gives the related answer.
It provides information about:
Courses
Fees
Admission process
Hostel
Library
Placement
Scholarship
WiFi
Sports and more
If the chatbot does not understand the question, it shows a default message.
When the user types “exit”, the chatbot thanks the user and stops execution.
