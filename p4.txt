class NqueenBranchBound {

    static int n = 4;
    static int board[][] = new int[n][n];

    // Arrays used for Branch and Bound
    static boolean column[] = new boolean[n];
    static boolean leftDiagonal[] = new boolean[2 * n];
    static boolean rightDiagonal[] = new boolean[2 * n];

    // Function to place queens
    static boolean Queen(int row) {

        // All queens placed
        if (row == n)
            return true;

        // Try every column
        for (int col = 0; col < n; col++) {

            // Check if position is safe
            if (!column[col] &&
                !leftDiagonal[row - col + n] &&
                !rightDiagonal[row + col]) {

                // Place queen
                board[row][col] = 1;

                // Mark column and diagonals
                column[col] = true;
                leftDiagonal[row - col + n] = true;
                rightDiagonal[row + col] = true;

                // Recursive call
                if (Queen(row + 1))
                    return true;

                // Backtracking
                board[row][col] = 0;

                // Unmark column and diagonals
                column[col] = false;
                leftDiagonal[row - col + n] = false;
                rightDiagonal[row + col] = false;
            }
        }

        return false;
    }

    public static void main(String[] args) {

        if (Queen(0)) {

            System.out.println("Solution Exists:\n");

            for (int i = 0; i < n; i++) {

                for (int j = 0; j < n; j++) {

                    if (board[i][j] == 1)
                        System.out.print(" Q ");
                    else
                        System.out.print(" _ ");
                }

                System.out.println();
            }

        } else {
            System.out.println("No Solution Exists");
        }
    }
}

Solves the N-Queens Problem on a 4×4 chessboard


Uses Constraint Satisfaction Problem (CSP) approach


Uses Backtracking algorithm


Uses Branch and Bound optimization


Places queens row by row


Checks safe position before placing queen


Avoids same column placement


Avoids same diagonal placement


If position is unsafe, tries another position


If no solution found, goes back and changes previous queen position


Continues until all queens are placed safely


Prints final valid chessboard arrangement


Q represents queen


_ represents empty space
