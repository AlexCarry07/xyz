import java.util.Scanner;

public class AdmissionChatbot {

    public static void main(String[] args) {

        Scanner sc = new Scanner(System.in);

        System.out.println("===== Welcome to Admission Chatbot =====");

        while (true) {

            System.out.print("\nAsk Question: ");
            String q = sc.nextLine().toLowerCase();

            if (q.contains("course")) {
                System.out.println("Available courses are CSE, IT, AIDS and ECE");
            }

            else if (q.contains("fees")) {
                System.out.println("Fees is 1,00,000 per year");
            }

            else if (q.contains("admission")) {
                System.out.println("Admission is based on entrance exam score");
            }

            else if (q.contains("hostel")) {
                System.out.println("Hostel facility is available for boys and girls");
            }

            else if (q.contains("contact")) {
                System.out.println("Contact: 9876543210");
            }

            else if (q.contains("library")) {
                System.out.println("Library facility is available from 8 AM to 8 PM");
            }

            else if (q.contains("placement")) {
                System.out.println("Top companies visit for placements every year");
            }

            else if (q.contains("canteen")) {
                System.out.println("Canteen facility is available inside campus");
            }

            else if (q.contains("scholarship")) {
                System.out.println("Scholarships are available for eligible students");
            }

            else if (q.contains("faculty")) {
                System.out.println("Experienced and qualified faculty members are available");
            }

            else if (q.contains("exam")) {
                System.out.println("Exams are conducted semester wise");
            }

            else if (q.contains("timing")) {
                System.out.println("College timing is 9 AM to 5 PM");
            }

            else if (q.contains("location")) {
                System.out.println("College is located in Pune");
            }

            else if (q.contains("transport")) {
                System.out.println("Bus transport facility is available");
            }

            else if (q.contains("wifi")) {
                System.out.println("Free WiFi facility is available on campus");
            }

            else if (q.contains("sports")) {
                System.out.println("Sports and gym facilities are available");
            }

            else if (q.contains("exit")) {
                System.out.println("Thank You for using Admission Chatbot");
                break;
            }

            else {
                System.out.println("Sorry, I don't understand");
            }
        }

        sc.close();
    }
}
