import java.util.*;

class SelectionSort {
    public static void main(String args[]) {

        Scanner sc = new Scanner(System.in);

        // Input number of elements
        System.out.println("Enter no of elem");
        int n = sc.nextInt();

        // Input array elements
        System.out.println("Enter elem");
        int a[] = new int[n];

        for (int i = 0; i < n; i++) {
            a[i] = sc.nextInt();
        }

        // Selection Sort
        for (int i = 0; i < n - 1; i++) {

            int min = i; // assume current index is minimum

            // Find smallest element
            for (int j = i + 1; j < n; j++) {
                if (a[j] < a[min]) {
                    min = j;
                }
            }

            // Swap elements
            int temp = a[i];
            a[i] = a[min];
            a[min] = temp;
        }
	System.out.print("selection sort : ");

        // Print sorted array
        for (int x : a) {
            System.out.print(x + " ");
        }
    }
}







































































































































































































































































































































































































































































































































































































































































































































































































































The program first asks the user to enter the number of elements in the array.
Then it takes all the array elements as input from the user.
After taking input, the Selection Sort algorithm starts.
How Selection Sort Works
The algorithm assumes the first element is the smallest.
It then compares this element with all remaining elements in the array.
If a smaller element is found, its position is stored.
After checking all elements, the smallest element is swapped with the current element.
This process repeats for every position in the array until the array becomes sorted.
Example

If the array is:
64 25 12 22 11

In the first pass, 11 is the smallest, so it is placed at the first position.
In the second pass, 12 is the smallest among remaining elements, so it is placed at the second position.
The process continues until all elements are arranged in ascending order.
Final Output

The program prints the sorted array in ascending order.
