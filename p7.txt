public class Calculator {

    public static void Cal() {

        Integer a = 40;
        Integer b = 20;

        Integer sum = a + b;
        Integer sub = a - b;

        System.debug('Sum = ' + sum);
        System.debug('Sub = ' + sub);
    }
}






Calculator.Cal();
